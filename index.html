const aggBy=(rows,field)=>{

  const g={};

  for(const r of rows){

    const k=r[field]||'Unknown';

    if(!g[k])g[k]={name:k,spend:0,impressions:0,clicks:0,reach:0,freq_sum:0,conversions:0,add_to_cart:0,initiate_checkout:0,purchases:0,purchase_value:0,messaging_conversations:0,custom_event:0,lead:0,cnt:0};

    g[k].spend+=r.spend||0;g[k].impressions+=r.impressions||0;g[k].clicks+=r.clicks||0;

    g[k].reach+=r.reach||0;g[k].freq_sum+=r.frequency||0;

    g[k].purchases+=r.purchases||0;g[k].purchase_value+=r.purchase_value||0;

    g[k].messaging_conversations+=r.messaging_conversations||0;

    g[k].custom_event+=r.custom_event||0;g[k].lead+=r.lead||0;g[k].cnt++;

  }

  return Object.values(g).map(x=>({

    ...x,

    ctr:x.impressions>0?x.clicks/x.impressions:0,

    cpm:x.impressions>0?(x.spend/x.impressions)*1000:0,

    cpc:x.clicks>0?x.spend/x.clicks:0,

    frequency:x.cnt>0?x.freq_sum/x.cnt:0,

    cost_per_custom:x.custom_event>0?x.spend/x.custom_event:0,

    roas:x.spend>0?x.purchase_value/x.spend:0,

    cost_per_cart:x.add_to_cart>0?x.spend/x.add_to_cart:0,

    cost_per_checkout:x.initiate_checkout>0?x.spend/x.initiate_checkout:0,

const timeSeries=(rows,field,names,gran,metric)=>{

  if(!names||!names.length)return{};

  const filtered=rows.filter(r=>names.includes(r[field]));

  const grp={};

  for(const r of filtered){

    let dk=r.date;

    if(gran==='weekly'){const d=new Date(r.date+'T00:00:00');d.setDate(d.getDate()-d.getDay()+1);dk=d.toISOString().split('T')[0];}

    else if(gran==='monthly')dk=r.date.slice(0,7);

    const k=`${r[field]}|${dk}`;

    if(!grp[k])grp[k]={e:r[field],d:dk,spend:0,impressions:0,clicks:0,fs:0,cnt:0,purchases:0,purchase_value:0,add_to_cart:0,initiate_checkout:0,messaging_conversations:0,custom_event:0};

    grp[k].spend+=r.spend||0;grp[k].impressions+=r.impressions||0;grp[k].clicks+=r.clicks||0;grp[k].fs+=r.frequency||0;grp[k].cnt++;

    grp[k].add_to_cart+=r.add_to_cart||0;grp[k].initiate_checkout+=r.initiate_checkout||0;

    grp[k].messaging_conversations+=r.messaging_conversations||0;grp[k].custom_event+=r.custom_event||0;

  }

  for(const k in grp){

    const g=grp[k];

    let v=0;

    switch(metric){

      case'spend':v=g.spend;break;case'impressions':v=g.impressions;break;case'clicks':v=g.clicks;break;

      case'ctr':v=g.impressions>0?(g.clicks/g.impressions)*100:0;break;

      case'cpm':v=g.impressions>0?(g.spend/g.impressions)*1000:0;break;

      case'cpc':v=g.clicks>0?g.spend/g.clicks:0;break;

      case'frequency':v=g.cnt>0?g.fs/g.cnt:0;break;

      case'purchases':v=g.purchases;break;

      case'custom_event':v=g.custom_event;break;

      default:v=g.spend;

    }

    if(!res[g.e])res[g.e]=[];

    res[g.e].push({date:g.d,value:+v.toFixed(2)});

  }

const SidebarAccountSelector=({label,icon,color,rgb,ids,names,selected,onChange})=>{

  const [q,setQ]=useState('');

  const allSelected=selected===null;

  const noneSelected=selected!==null&&selected.length===0;

  const selArr=selected===null?ids:(selected||[]);

  const filtered=q.trim()?ids.filter(id=>(names[id]||id).toLowerCase().includes(q.toLowerCase())):ids;

  const allFilteredChecked=filtered.every(id=>selArr.includes(id));


  const toggle=id=>{

    const cur=selected===null?[...ids]:selected;

    const next=cur.includes(id)?cur.filter(x=>x!==id):[...cur,id];

    onChange(next.length===ids.length?null:next);

  };

  const selectAll=()=>onChange(null);

  const deselectAll=()=>onChange([]);

  const toggleFiltered=()=>{

    if(allFilteredChecked){

      // deselect only the filtered ones

      const cur=selected===null?[...ids]:selected;

      const next=cur.filter(id=>!filtered.includes(id));

      onChange(next.length===0?[]:next.length===ids.length?null:next);

    } else {

      // select all filtered

      const cur=selected===null?[...ids]:selected;

      const next=[...new Set([...cur,...filtered])];

      onChange(next.length===ids.length?null:next);

    }

  };


  return(

    <div style={{padding:'0 8px 10px'}}>

      {/* Header */}

        <div style={{position:'relative',marginBottom:'4px'}}>

          <input value={q} onChange={e=>setQ(e.target.value)} placeholder="Buscar conta…"

            style={{width:'100%',padding:'4px 22px 4px 7px',background:'#0a0d14',border:'1px solid #1f2937',borderRadius:'5px',color:'#f1f5f9',fontSize:'11px',outline:'none',boxSizing:'border-box'}}/>

      }

      if(event==='SIGNED_OUT')setAuthState({screen:'login',user:null});

      if(event==='TOKEN_REFRESHED'&&session){

        setAuthState(prev=>prev.screen==='app'?{...prev,user:{...prev.user,_token:session.access_token}}:prev);

      }

    });

    return()=>subscription.unsubscribe();

  },[]);

  const handleOnboard=u=>setAuthState({screen:'app',user:u});
